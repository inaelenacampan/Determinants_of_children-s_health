 16def bar_plot(year, dfs, variable, guide):
 17    """
 18    Réaliser un bar plot.
 19
 20    Args :
 21        year (str) : années d'intéret
 22        dfs (dict) : dictionnaire des bases de données
 23        variable (str) : Le critère de regroupement.
 24        guide (pd object) : Le guide des variables NSCH.
 25
 26    Returns : un bar plot
 27    """
 28
 29    df = dfs[year]
 30    ax = sns.countplot(data=df, x=variable)
 31
 32    # récuperer la signification du codage de la réponse
 33    responde_str = guide.loc[guide["Variable"] == variable, "Response Code"].iloc[0]
 34    code_dict = dict(item.split(' = ') for item in responde_str.split('||'))
 35    question_text = guide.loc[guide["Variable"] == variable, "Question"].iloc[0]
 36
 37    # eviter l'affichage d'un warning lié au ticks
 38    ticks = ax.get_xticks()
 39    old_labels = [label.get_text() for label in ax.get_xticklabels()]
 40
 41    # on utilise le dictionnaire pour un affichage lisible
 42    new_labels = [code_dict[name] for name in old_labels]
 43    ax.set_xticks(ticks)
 44    ax.set_xticklabels(new_labels)
 45
 46    # pivoter les labels
 47    plt.xticks(rotation=45, fontsize=8)
 48    plt.xlabel("Réponse")
 49    plt.ylabel("Effectif")
 50    plt.title(f"Répartition de la variable d'intérêt \n{question_text}", fontsize=8)
 51    plt.show()

 90def mca_analysis(year, dfs, drop_columns):
 91    """
 92    Réaliser une ACM.
 93
 94    Args:
 95        year (str): L'année du formulaire.
 96        dfs (dict): Le dictionnaire des bases des formulaires NSCH.
 97        drop_columns (list) : La liste des colonnes à filtrer.
 98
 99    Returns:
100        df_mca : La base de données adéquate pour réaliser l'ACM.
101        mca : Objet mca.
102    """
103    df = dfs[year]
104    df_mca = df.drop(columns=[col for col in drop_columns])
105    df_mca = df_mca.drop(columns=[col for col in df_mca.columns if 'imputed' in col])
106
107    mca = prince.MCA(n_components=2).fit(df_mca)
108    return df_mca, mca

148def mca_plot_individuals_group(df_mca, mca, variable, guide):
149    """
150    Visualisation des individus selon un critère de regroument.
151
152    Args:
153        df_mca : La base de données adéquate pour réaliser l'ACM.
154        mca : L'objet mca.
155        variable (str) : Le critère de regroupement.
156        guide (pd object) : Le guide des variables NSCH.
157
158    Returns:
159        génération d'un plot (pas de return explicite)
160    """
161    row_coords = mca.row_coordinates(df_mca)
162    groups = df_mca[variable].unique()
163    # Les variables sont catégorielles et la légende se présente sous forme d'un str
164    # format : ciffre1 = ... || chiffre2 = ...
165    responde_str = guide.loc[guide["Variable"] == variable, "Response Code"].iloc[0]
166    code_dict = dict(item.split(' = ') for item in responde_str.split('||'))
167
168    plt.figure(figsize=(10, 10))
169    palette = sns.color_palette("tab10", len(groups))
170
171    for i, group in enumerate(groups):
172        mask = (df_mca[variable] == group)
173        x = row_coords[0][mask]
174        y = row_coords[1][mask]
175        # Scatter plot par groupe
176        label = code_dict.get(str(group), str(group))
177        plt.scatter(x, y, s=5, alpha=0.6, color=palette[i], label=label)
178        # code adapté depuis internet pour manipuler les ellipses
179        # Ellipse pour montrer un regroupement des individus
180        # plus d'un individu
181        if len(x) > 1:
182            cov = np.cov(x, y)
183            # calcule les valeurs propres (vals) et vecteurs propres (vecs) d’une matrice symétrique
184            vals, vecs = np.linalg.eigh(cov)
185            width, height = 2 * np.sqrt(vals) * 2  # facteur 2 pour agrandir ellipse
186            angle = np.degrees(np.arctan2(*vecs[:, 0][::-1]))
187            ell = Ellipse(xy=(np.mean(x), np.mean(y)),
188                          width=width, height=height,
189                          angle=angle,
190                          color=palette[i], alpha=0.2)
191            # GCA = Get Current Axes
192            plt.gca().add_patch(ell)
193
194    # Récupération du texte de la question pour le titre
195    question_text = guide.loc[guide["Variable"] == variable, "Question"].iloc[0]
196
197    plt.xlabel('Dimension 0', fontsize=12)
198    plt.ylabel('Dimension 1', fontsize=12)
199
200    plt.title(f'ACM des individus par groupe: \n{question_text}', fontsize=10)
201    plt.grid()
202
203    plt.legend()
204    plt.show()

207def mca_plot_categories(df_mca, mca, seuil):
208    """
209    Visualisation des modalités (variables catégorielles) de la base de données dans un plan 2D.
210
211    Args:
212        df_mca : La base de données adéquate pour réaliser l'ACM.
213        mca : L'objet mca.
214        seuil (int) : Pour mieux visualiser le graphique, combien de composantes "principales".
215
216    Returns:
217        génération d'une figure avec 2 plot (pas de return explicite)
218    """
219    # Pour les modalités, le graphique est très dense
220    # La longueur de la flèche indique l’importance de cette modalité dans l’espace MCA
221    # plus la flèche est longue, plus cette modalité contribue à la variance
222    # des premières dimensions.
223    column_coords = mca.column_coordinates(df_mca)
224
225    distances = np.sqrt(column_coords[0]**2 + column_coords[1]**2)
226    top = distances.sort_values(ascending=False).head(seuil).index
227    principales = column_coords.loc[top]
228    secondaires = column_coords.drop(top)
229
230    fig, axes = plt.subplots(1, 2, figsize=(24, 12))  # 1 ligne, 2 plots
231
232    # ---------------------------
233    # Plot n°1
234    # ---------------------------
235
236    ax = axes[0]
237    for i, col in enumerate(principales.index):
238        x, y = principales.iloc[i, 0], principales.iloc[i, 1]
239        ax.arrow(0, 0, x, y, color='red', alpha=0.7, head_width=0.02)
240        ax.text(x*1.05, y*1.05, col, color='red', fontsize=10)
241
242    ax.set_xlabel('Dimension 0')
243    ax.set_ylabel('Dimension 1')
244    ax.set_title('Modalités principales')
245    ax.grid()
246
247    # ---------------------------
248    # Plot n°2
249    # ---------------------------
250    ax = axes[1]
251    for i, col in enumerate(secondaires.index):
252        x, y = secondaires.iloc[i, 0], secondaires.iloc[i, 1]
253        ax.arrow(0, 0, x, y, color='red', alpha=0.3, head_width=0.01)
254        ax.text(x*1.05, y*1.05, col, color='red', fontsize=8)
255
256    ax.set_xlabel('Dimension 0')
257    ax.set_ylabel('Dimension 1')
258    ax.set_title('Modalités restantes')
259    ax.grid()
260
261    plt.tight_layout()
262    plt.show()

268
269    Args:

271
272    Returns:
273        génération d'un plot (pas de return explicite)

333def map_united_states(df_indicator, df_geo, year):
334    """
335    Crée une carte choroplèthe interactive des États-Unis représentant
336    un indicateur global de santé des enfants pour une année donnée.
337
338    La carte est générée à l’aide de Folium à partir de données géographiques
339    (GeoDataFrame) et d’un DataFrame contenant les indicateurs de santé.
340    Chaque État est coloré selon la valeur de l’indicateur, allant du rouge
341    (faible niveau de santé) au vert foncé (meilleur niveau de santé).
342
343    Args:
344        df_indicator : pandas.DataFrame
345            DataFrame contenant l'indicateur de santé par État.
346            L'index doit correspondre au code FIPS des États (FIPSST).
347            Il doit contenir une colonne nommée :
348            "indicator_global_health_<year>".
349
350        df_geo : geopandas.GeoDataFrame
351            GeoDataFrame contenant les polygones des États américains.
352            Doit inclure les colonnes :
353            - "GeoFIPS" : code FIPS sous forme de chaîne
354            - "GeoName" : nom de l'État
355            - "geometry" : géométrie des États
356
357        year : str
358            Année de l'indicateur de santé à afficher sur la carte.
359
360    Returns:
361        folium.Map
362            Objet Folium représentant la carte interactive des États-Unis,
363            avec une légende et des infobulles affichant le nom de l'État
364            et la valeur de l'indicateur de santé.
365
366    Références
367    ----------
368    Tutoriel Folium utilisé :
369    https://python-visualization.github.io/folium/latest/user_guide/geojson/geojson_popup_and_tooltip.html
370    """
371
372    new_df = df_geo.copy()
373    new_df["FIPSST"] = new_df["GeoFIPS"].str.replace('"', '').str.strip().str[:-3].astype(int)
374
375    gdf = gpd.GeoDataFrame(new_df, geometry='geometry')
376    df_indicator = df_indicator.rename_axis('FIPSST')
377
378    # Merge pour ajouter l'indicateur dans gdf
379    gdf = gdf.merge(df_indicator, on="FIPSST", how="left")
380
381    # Centrer la carte sur les USA
382    m = folium.Map([43, -100], zoom_start=4, min_zoom=3, max_zoom=6)
383
384    # Créer une échelle de couleur propre : rouge ---> vert
385    colormap = branca.colormap.LinearColormap(
386        vmin=gdf[f"indicator_global_health_{year}"].min(),
387        vmax=gdf[f"indicator_global_health_{year}"].max(),
388        colors=["red", "orange", "lightblue", "green", "darkgreen"],
389        caption=f"Indice global de santé (0-1) pour l'année {year}"
390    )
391
392    # ajout des polygones et des messages
393    folium.GeoJson(
394        gdf,
395        style_function=lambda feature: {
396            'fillColor': colormap(feature['properties'][f'indicator_global_health_{year}']),
397            'color': 'black',
398            'weight': 0.5,
399            'fillOpacity': 0.8,
400        },
401        tooltip=folium.GeoJsonTooltip(
402            fields=["GeoName", f"indicator_global_health_{year}"],
403            aliases=["État:", "Indice santé :"],
404            localize=True
405        )
406    ).add_to(m)
407
408    # Ajouter la légende du colormap
409    colormap.add_to(m)
410
411    return m

461def kendall_analysis(df_indicator):
462    """
463    Calcule le taux de Kendall (tau) entre l'indicateur global de santé et
464    plusieurs sous-indicateurs pour chaque année, et retourne un DataFrame récapitulatif.
465
466    Args:
467        df_indicator : pandas.DataFrame
468            DataFrame contenant les colonnes suivantes pour chaque année :
469            - indicator_global_health_{year}
470            - sub_indicator_eco_{year}
471            - sub_indicator_health_{year}
472            - sub_indicator_mental_{year}
473
474    Returns:
475        pandas.DataFrame
476            DataFrame contenant pour chaque sous-indicateur et chaque année :
477            - Year : l'année
478            - Sub_indicator : le nom du sous-indicateur
479            - Kendall_tau : le taux de Kendall entre le sous-indicateur et l'indicateur global
480            - p_value : la p-value du test de Kendall
481    """
482
483    years = ["2024", "2023", "2022", "2021"]
484    sub_indicators = ["sub_indicator_eco", "sub_indicator_health", "sub_indicator_mental"]
485
486    results = []
487
488    for year in years:
489        for sub in sub_indicators:
490            # Calcul du tau de Kendall et de la p-value pour tester la signification statistique
491            df1 = df_indicator[f"indicator_global_health_{year}"].sort_values(ascending=True)
492            df_global = df1.index.tolist()
493
494            df2 = df_indicator[f"{sub}_{year}"].sort_values(ascending=True)
495            df_sub = df2.index.tolist()
496
497            tau, _ = kendalltau(
498                df_global, df_sub
499            )
500
501            results.append({
502                "Year": year,
503                "Sub_indicator": sub,
504                "Kendall_tau": tau
505            })
506
507    # Retourner un DataFrame
508    return pd.DataFrame(results)

511def state_rankings(df_indicator, df_geo):
512    """
513    Génère et sauvegarde une heatmap représentant le classement des États
514    américains selon un indicateur global de santé pour plusieurs années.
515
516    La fonction associe les scores de santé contenus dans `df_indicator`
517    aux noms des États présents dans `df_geo`, calcule les classements
518    annuels (1 = meilleur État), puis affiche une matrice colorée où les
519    lignes correspondent aux États et les colonnes aux années. Les États
520    sont triés selon leur classement pour l'année la plus récente (2024).
521
522    Le graphique utilise une échelle de couleurs allant du vert (meilleur
523    classement) au rouge (moins bon classement), avec les années affichées
524    en haut du tableau.
525
526    Args:
527        df_indicator : pandas.DataFrame
528            Table contenant les indicateurs globaux de santé par État, indexée
529            par le code FIPS des États (FIPSST). Les colonnes doivent suivre la
530            convention de nommage :
531            `indicator_global_health_<année>`.
532
533        df_geo : pandas.DataFrame
534            Table de correspondance géographique contenant les informations
535            des États. Doit inclure les colonnes :
536            - `GeoFIPS` : code FIPS géographique de l'État
537            - `GeoName` : nom de l'État
538
539    Returns:
540    None
541        La fonction affiche le graphique et l'enregistre au format JPG
542        dans le dossier `docs/`.
543    """
544
545    df = df_indicator.reset_index()
546    new_df = df_geo.copy()
547    new_df["FIPSST"] = new_df["GeoFIPS"].str.replace('"', '').str.strip().str[:-3].astype(int)
548    df = df.merge(
549        new_df[['FIPSST', 'GeoName']],
550        on='FIPSST',
551        how='left'
552    )
553    df = df.set_index('GeoName')
554    years = [2024, 2023, 2023, 2021]
555    cols = [f'indicator_global_health_{y}' for y in years]
556
557    df_scores = df[cols]
558    df_rank = df_scores.rank(
559        axis=0,
560        ascending=False,   # Meilleur état - score le plus grand
561        method='min'
562    )
563    # renommer les colonnes

565
566    # classement de référence : l'année 2024
567    df_rank = df_rank.sort_values(by='2024')
568
569    plt.figure(figsize=(15, 30))
570    ax = sns.heatmap(
571        df_rank,
572        annot=True,
573        fmt=".0f",
574        cmap="RdYlGn_r",   # échelle : rouge (pire) ---> vert (meilleur)
575        cbar_kws={'label': 'Classement'}
576    )
577
578    # années affichées en haut du plot
579
580    ax.xaxis.tick_top()
581    ax.xaxis.set_label_position('top')
582    ax.set_xlabel("Année")
583
584    plt.title("Classement santé globale par État")
585    plt.ylabel("État")
586
587    # afficher la légénde de telle sorte que le vert soit en haut et le rouge en bas
588    cbar = ax.collections[0].colorbar
589    cbar.ax.invert_yaxis()
590
591    plt.tight_layout()
592
593    # sauvegarde au format jpg ---> inclusion dans le readMe
594    plt.savefig(
595        "docs/classement_etats_sante.jpg",
596        format="jpg",
597        dpi=300,
598        bbox_inches="tight"
599    )

Génère et sauvegarde une heatmap représentant le classement des États américains selon un indicateur global de santé pour plusieurs années.

La fonction associe les scores de santé contenus dans df_indicator aux noms des États présents dans df_geo, calcule les classements annuels (1 = meilleur État), puis affiche une matrice colorée où les lignes correspondent aux États et les colonnes aux années. Les États sont triés selon leur classement pour l'année la plus récente (2024).

Le graphique utilise une échelle de couleurs allant du vert (meilleur classement) au rouge (moins bon classement), avec les années affichées en haut du tableau.
